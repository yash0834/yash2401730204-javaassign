import java.util.Scanner;

class Calculator {
    
    public int add(int a, int b) {
        return a + b;
    }
    
    
    public double add(double a, double b) {
        return a + b;
    }
    
        public int add(int a, int b, int c) {
        return a + b + c;
    }
    
    
    public int subtract(int a, int b) {
        return a - b;
    }
    
   
    public double multiply(double a, double b) {
        return a * b;
    }
    
        public double divide(int a, int b) {
        if (b == 0) {
            System.out.println("Division by zero is not allowed!");
        }
        return (double) a / b;
    }
}

class UserInterface {
    private Scanner sc;
    
    public UserInterface() {
        sc = new Scanner(System.in);
    }
    
    
    public void performAddition(Calculator calc) {
        System.out.println("\n--- Addition ---");
        System.out.println("Choose addition type:");
        System.out.println("1. Add two integers");
        System.out.println("2. Add two doubles");
        System.out.println("3. Add three integers");
        System.out.print("Enter your choice (1-3): ");
        
        int choice = sc.nextInt();
        switch (choice) {
            case 1:
                System.out.print("Enter first integer: ");
                int int1 = sc.nextInt();
                System.out.print("Enter second integer: ");
                int int2 = sc.nextInt();
                System.out.println("Result: " + calc.add(int1, int2));
                break;
            case 2:
                System.out.print("Enter first double: ");
                double dbl1 = sc.nextDouble();
                System.out.print("Enter second double: ");
                double dbl2 = sc.nextDouble();
                System.out.println("Result: " + calc.add(dbl1, dbl2));
                break;
            case 3:
                System.out.print("Enter first integer: ");
                int i1 = sc.nextInt();
                System.out.print("Enter second integer: ");
                int i2 = sc.nextInt();
                System.out.print("Enter third integer: ");
                int i3 = sc.nextInt();
                System.out.println("Result: " + calc.add(i1, i2, i3));
                break;
            default:
                System.out.println("Invalid choice!");
        }
    }
    
        public void performSubtraction(Calculator calc) {
        System.out.println("\n--- Subtraction ---");
        System.out.print("Enter first integer: ");
        int a = sc.nextInt();
        System.out.print("Enter second integer: ");
        int b = sc.nextInt();
        System.out.println("Result: " + calc.subtract(a, b));
    }
    
    
    public void performMultiplication(Calculator calc) {
        System.out.println("\n--- Multiplication ---");
        System.out.print("Enter first double: ");
        double a = sc.nextDouble();
        System.out.print("Enter second double: ");
        double b = sc.nextDouble();
        System.out.println("Result: " + calc.multiply(a, b));
    }
    
    
    public void performDivision(Calculator calc) {
        System.out.println("\n--- Division ---");
        System.out.print("Enter numerator (integer): ");
        int a = sc.nextInt();
        System.out.print("Enter denominator (integer): ");
        int b = sc.nextInt();
        
        try {
            double result = calc.divide(a, b);
            System.out.println("Result: " + result);
        } catch (ArithmeticException e) {
            System.out.println("Error: " + e.getMessage());
        }
    }
    
        public void mainMenu(Calculator calc) {
        int choice;
        do {
            System.out.println("\nWelcome to the Calculator Application!");
            System.out.println("1. Add Numbers");
            System.out.println("2. Subtract Numbers");
            System.out.println("3. Multiply Numbers");
            System.out.println("4. Divide Numbers");
            System.out.println("5. Exit");
            System.out.print("Enter your choice: ");
            
            choice = sc.nextInt();
            
            switch (choice) {
                case 1:
                    performAddition(calc);
                    break;
                case 2:
                    performSubtraction(calc);
                    break;
                case 3:
                    performMultiplication(calc);
                    break;
                case 4:
                    performDivision(calc);
                    break;
                case 5:
                    System.out.println("Thank you for using the Calculator!");
                    break;
                default:
                    System.out.println("Invalid choice! Please try again.");
            }
        } while (choice != 5);
        
        sc.close();
    }
    
        public static void main(String[] args) {
        Calculator calc = new Calculator();
        UserInterface ui = new UserInterface();
        ui.mainMenu(calc);
    }
}
